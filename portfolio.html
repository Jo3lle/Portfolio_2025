  • ðŸ”¹ Zodra de knop werd ingedrukt, schakelde de LED aan, en als de knop werd losgelaten, ging de LED uit.
Codevoorbeeld

                    int buttonPin = 2; // Drukknop aangesloten op pin 2
                    int ledPin = 13;   // LED aangesloten op pin 13
                    
                    void setup() {
                        pinMode(buttonPin, INPUT);
                        pinMode(ledPin, OUTPUT);
                    }
                    
                    void loop() {
                        int buttonState = digitalRead(buttonPin);
                        if (buttonState == HIGH) {
                            digitalWrite(ledPin, HIGH); // LED aan
                        } else {
                            digitalWrite(ledPin, LOW);  // LED uit
                        }
                    }
